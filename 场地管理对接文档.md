## 场地管理相关

### 获取首页查询表单字段（Get）

- 请求地址

  ```
  <项目地址>/config/dynamicFields
  ```

- 请求参数

  无

- 返回参数

  > 返回参数中radioOpts即为查询字段。其中，第一级字段label代表类型，不同label对应不同的options字段；
  >
  > options中共有三种输入项，inputOpts对应文本框；radioOpts对应单选项；checkboxOpts对应多选项；
  >
  > 第二级label为输入项的名称；field为请求form的name属性；value为输入值

  

  ```js
  {
    code: "S_OK",
    data: {
      message: {
        result:"success",
      	radioOpts: [
          {
            label: "酒店",//类型，不同类型对应不同options
            options: {
              //文本框
              inputOpts: [
                {
                  field: "beachNum",
                  label: "沙滩数量",
                  value: "0"
                }
              ],
              //单选项
              radioOpts: [
                {
                  field: "field5",
                  label: "单选1",
                  value: "",
                  radios: [
                    {
                      label: "选项1"
                    },
                    {
                      label: "选项2"
                    },
                    {
                      label: "选项3"
                    }
                  ]
                },
                {
                  field: "field6",
                  label: "单选2",
                  value: "",
                  radios: [
                    {
                      label: "选项1"
                    },
                    {
                      label: "选项2"
                    },
                    {
                      label: "选项3"
                    }
                  ]
                }
              ],
              checkboxOpts: [
                {
                  field: "field7",
                  label: "多选1",
                  value: ["选项1"],
                  checkboxs: [
                    {
                      label: "选项1"
                    },
                    {
                      label: "选项2"
                    },
                    {
                      label: "选项3"
                    }
                  ]
                },
                {
                  field: "field8",
                  label: "多选2",
                  value: ["选项1"],
                  checkboxs: [
                    {
                      label: "选项1"
                    },
                    {
                      label: "选项2"
                    },
                    {
                      label: "选项3"
                    }
                  ]
                }
              ]
            }
          },
          {
            label: "美食",
            options: {
              inputOpts: [
                {
                  field: "field1",
                  label: "卡路里上限",
                  value: "0"
                },
                {
                  field: "field2",
                  label: "好评度",
                  value: "0"
                },
                {
                  field: "field3",
                  label: "月销量",
                  value: "0"
                }
              ]
            }
          },
          {
            label: "旅馆",
            options:{}
          },
          {
            label: "游泳",
            options:{}
          }
        ]
      }
    }
  }
  ```

  

### 搜索查询场地(Get)

- 请求地址

  ```
  <项目地址>/config/indexSearchResults
  ```

- 请求参数

  > 键名由获取首页查询表单字段的接口返回的field属性获取

  

  ```js
  {
    ballroomArea: "0232"//文本框
    beachNum: "023"//文本框
    field1: "02323"//文本框
    //...
    field5: '2131'//单选选项;根据value返回
    field6:123//单选选项;根据value返回
    field7: ["选项1", "选项3"]//多选选项;数组
    field8:  ["选项1", "选项3", "选项2"]//多选选项;数组
  }
  ```

- 返回参数

  ```js
  {
    code: "S_OK",
    data: {
      message: {
        result:"success",
      	data:
        [
          {
            pic: "http://store.is.autonavi.com/showpic/6758bc1520cb4f61b67ac6b1413cf26a?operate=merge&w=160&h=150&position=5",//图片
            name: "凯宾斯基酒店",//名称
            type: "酒店",//搜索类型
            id: "B0FFFFN5HZ",//id
            location: {//经纬度
              lat: "116.481565",
              lng: "39.989449"
            }
            //以上为固定属性；以下字段可根据不同类型而不同；但必须包含label和value两个参数指明显示项名称及显示值
            price: {
              label: "价格",
              value: "￥200/晚"	
            },
            address: {
              label: "地址",
              value: "综合商业南区A口500米处"
            },
          },
          {
            pic: "http://store.is.autonavi.com/showpic/d829b8ed2bb74b3e56ceac85b4b66aa5?operate=merge&w=160&h=150&position=5",
            name: "XXX酒店",
            type: "酒店",
            id: "B000A7RYT5",
            location: {
              lat: "116.484401",
              lng: "39.989934"
            }
            //以上为固定属性；以下字段可根据不同类型而不同；但必须包含label和value两个参数指明显示项名称及显示值
            price: {
              label: "价格",
              value: "￥100/晚"
            },
            area: {
              label: "面积",
              value: "75m^2"
            }
          }
        ]
      }
    }
  }
  ```
